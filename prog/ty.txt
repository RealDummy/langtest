struct foo {
    a,
    b,
}

let single = foo()

single.a = 2
single.b = 4

struct bar {
    arra,
    arrb,
}


func arrify(n) {
    let na = []
    let i = 0
    while i < n {
        na + n
        i = i + 1
    }
    return na
}

let arr = bar()

arr.arra = arrify(single.a)
arr.arrb = arrify(single.b)

print single
print arr
